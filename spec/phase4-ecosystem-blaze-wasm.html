Blaze Phase 4 – Ecosystem: Wasm

Previous: Wasm Bindgen | Next: Web

Goal

WebAssembly runtime embedded in Blaze: load, compile, instantiate, and execute .wasm modules. Supports MVP, sign‑extension, mutable globals, SIMD.

1. Core Types

pub struct Engine { /* … */ }
pub struct Module { /* compiled */ }
pub struct Instance { module: Module, memory: Option<Memory>, exports: Exports }
pub struct Memory { buf: Vec<u8>, max_pages: u32 }
pub struct Func { /* … */ }
pub enum Value { I32(i32), I64(i64), F32(f32), F64(f64), V128(u128) }

2. Engine

impl Engine {
    pub fn new(config: &EngineConfig) -> Self;
    pub fn compile(&self, wasm_bytes: &[u8]) -> Result<Module, WasmError>;
    pub fn instantiate(&self, module: &Module, imports: &Imports) -> Result<Instance, WasmError>;
}

3. Imports

pub struct Imports {
    functions: Map<(Text, Text), Func>,
    memories: Map<(Text, Text), Memory>,
    globals: Map<(Text, Text), Global>,
    tables: Map<(Text, Text), Table>,
}
impl Imports {
    pub fn new() -> Self;
    pub fn func(mut self, module: &str, name: &str, f: Func) -> Self;
    pub fn memory(mut self, module: &str, name: &str, m: Memory) -> Self;
    pub fn done(self) -> Self;
}

4. Instance

impl Instance {
    pub fn get_export(&self, name: &str) -> Option<&Extern>;
    pub fn memory(&self) -> Option<&Memory>;
    pub fn call(&self, func_name: &str, args: &[Value]) -> Result<Vec<Value>, WasmError>;
}
pub enum Extern { Func(Func), Memory(Memory), Global(Global), Table(Table) }

5. Host Functions

impl Func {
    pub fn new<F>(f: F) -> Self where F: Fn(&[Value]) -> Result<Vec<Value>, WasmError> + Send + 'static;
    pub fn call(&self, args: &[Value]) -> Result<Vec<Value>, WasmError>;
}

6. Error Handling

pub enum WasmError {
    Compile(Text), Instantiate(Text), Runtime(RuntimeError),
    MemoryOutOfBounds, Unreachable, TypeMismatch(Text), ImportNotFound(Text),
}
pub struct RuntimeError { message: Text, function: Option<Text>, wasm_stack_trace: Vec<Frame> }
