Goal

TLS client and server built on native library (OpenSSL/BoringSSL). Async streams, certificate handling, ALPN.

1. Configuration

pub struct TlsConfig {
    cert_file: Option<Text>,
    key_file: Option<Text>,
    ca_file: Option<Text>,
    verify_hostname: bool,
    min_protocol_version: ProtocolVersion,
    max_protocol_version: ProtocolVersion,
    cipher_list: Option<Text>,
    alpn_protocols: Vec<Text>,
    enable_sni: bool,
}
pub enum ProtocolVersion { TlsV1_0, TlsV1_1, TlsV1_2, TlsV1_3 }

5. Error Handling

pub enum TlsError {
    Io(std::io::Error),
    Config(Text),
    Handshake(HandshakeError),
    Certificate(Text),
    Protocol(Text),
    Timeout,
}
pub struct HandshakeError { kind: HandshakeErrorKind, message: Text }
pub enum HandshakeErrorKind { NoCertificates, InvalidCertificate, UnknownCa, HandshakeFailure, ProtocolVersion, BadRecordMac, DecodeError, UnexpectedMessage }
