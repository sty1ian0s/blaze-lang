Goal

Lightweight, single‑threaded async executor for embedded, WASM, and minimal runtime scenarios. #![no_std]‑friendly.

1. LocalExecutor

pub struct LocalExecutor { /* task queue, timer wheel */ }

    pub fn block_on<F: Future>(&self, future: F) -> F::Output;
    pub fn run_until_stalled(&self);
}
