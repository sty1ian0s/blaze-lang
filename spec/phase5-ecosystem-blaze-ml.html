Blaze Phase 5 – Ecosystem: ML

Previous: Dataframe | Next: Tensor

Goal

Data‑oriented, zero‑cost machine learning framework with automatic differentiation, gradient‑based optimisation, and GPU offloading.

1. Core Types

pub struct Tensor<T, const N: usize> { /* … */ } // see blaze-tensor
pub struct Grad<T: Numeric> { data: Tensor<T, 1>, gradient: Option<Tensor<T, 1>>, tape: Option<Tape> }
pub trait Module<T: Numeric> {
    fn forward(&self, input: &Tensor<T, 2>) -> Tensor<T, 2>;
    fn parameters(&self) -> Vec<Grad<T>>;
}
pub trait Optimizer<T: Numeric> {
    fn step(&mut self, parameters: &mut [Grad<T>]);
    fn zero_grad(&mut self);
}
pub trait Loss<T: Numeric> { fn forward(&self, predicted: &Tensor<T, 2>, target: &Tensor<T, 2>) -> T; }

2. Optimizers

pub struct SGD<T> { learning_rate: T }
pub struct Adam<T> { learning_rate: T, beta1: T, beta2: T, epsilon: T }

3. Loss Functions

pub struct MseLoss<T>; pub struct CrossEntropyLoss<T>; pub struct BinaryCrossEntropyLoss<T>;

4. Data Loading

pub trait Dataset<T> { fn len(&self) -> usize; fn get(&self, index: usize) -> (Tensor<T, 2>, Tensor<T, 2>); }
pub struct DataLoader<D: Dataset<T>, T> { dataset: D, batch_size: usize, shuffle: bool }
impl<D: Dataset<T>, T> Iterator for DataLoader<D, T> { type Item = (Tensor<T, 2>, Tensor<T, 2>); }

5. Serialization

pub fn save<M: Module<f32>>(module: &M, path: &str) -> Result<(), MlError>;
pub fn load<M: Module<f32> + 'static>(path: &str) -> Result<M, MlError>;

6. Error Handling

pub enum MlError {
    ShapeMismatch(Text), DeviceMismatch, OutOfMemory,
    InvalidParameter(Text), LossNotReduced, LoadError(Text), SaveError(Text),
}
