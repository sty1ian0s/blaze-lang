Blaze Phase 4 – Ecosystem: Vulkan

Previous: TOML | Next: Wasm Bindgen

Goal

Low‑level, unsafe, data‑oriented bindings to Vulkan 1.3. All resources are linear (Instance, Device, Buffer, Image, CommandBuffer, etc.).

1. Core Types

pub struct Instance { handle: vk::Instance, entry: Entry }
pub struct PhysicalDevice { handle: vk::PhysicalDevice }
pub struct Device { handle: vk::Device, queues: Map<u32, Queue> }
pub struct Queue { handle: vk::Queue, family: u32 }
pub struct DeviceMemory { handle: vk::DeviceMemory, size: vk::DeviceSize }
pub struct Buffer { handle: vk::Buffer, memory: DeviceMemory, size: vk::DeviceSize }
pub struct Image { handle: vk::Image, memory: DeviceMemory, format: vk::Format, extent: vk::Extent3D }
/* … and many more: ImageView, Sampler, Framebuffer, RenderPass, Pipeline, ShaderModule, DescriptorSetLayout, DescriptorPool, DescriptorSet, CommandPool, CommandBuffer, Fence, Semaphore, Event, QueryPool */

2. Instance

impl Instance { pub fn new(create_info: &InstanceCreateInfo) -> Result<Self, VkError>; pub fn enumerate_physical_devices(&self) -> Vec<PhysicalDevice>; }

3. Device

impl PhysicalDevice {
    pub fn create_device(&self, create_info: &DeviceCreateInfo) -> Result<Device, VkError>;
}
impl Device {
    pub fn allocate_memory(&self, allocate_info: &MemoryAllocateInfo) -> Result<DeviceMemory, VkError>;
    pub fn create_buffer(&self, create_info: &BufferCreateInfo) -> Result<Buffer, VkError>;
    pub fn create_image(&self, create_info: &ImageCreateInfo) -> Result<Image, VkError>;
    pub fn create_command_pool(&self, create_info: &CommandPoolCreateInfo) -> Result<CommandPool, VkError>;
    pub fn create_command_buffer(&self, allocate_info: &CommandBufferAllocateInfo) -> Vec<CommandBuffer>;
}

4. Command Recording

impl CommandBuffer {
    pub fn begin(&mut self, info: &CommandBufferBeginInfo);
    pub fn end(&mut self);
    pub fn cmd_begin_render_pass(&mut self, info: &RenderPassBeginInfo, contents: SubpassContents);
    pub fn cmd_end_render_pass(&mut self);
    pub fn cmd_bind_pipeline(&mut self, bind_point: PipelineBindPoint, pipeline: &Pipeline);
    pub fn cmd_bind_vertex_buffers(&mut self, first_binding: u32, buffers: &[&Buffer], offsets: &[vk::DeviceSize]);
    pub fn cmd_bind_index_buffer(&mut self, buffer: &Buffer, offset: vk::DeviceSize, index_type: IndexType);
    pub fn cmd_bind_descriptor_sets(&mut self, bind_point: PipelineBindPoint, layout: &PipelineLayout, first_set: u32, descriptor_sets: &[&DescriptorSet], dynamic_offsets: &[u32]);
    pub fn cmd_draw(&mut self, vertex_count: u32, instance_count: u32, first_vertex: u32, first_instance: u32);
    pub fn cmd_draw_indexed(&mut self, index_count: u32, instance_count: u32, first_index: u32, vertex_offset: i32, first_instance: u32);
    pub fn cmd_dispatch(&mut self, group_count_x: u32, group_count_y: u32, group_count_z: u32);
    pub fn cmd_copy_buffer(&mut self, src: &Buffer, dst: &Buffer, regions: &[BufferCopy]);
    pub fn cmd_pipeline_barrier(&mut self, src_stage: PipelineStageFlags, dst_stage: PipelineStageFlags, dependency_flags: DependencyFlags, memory_barriers: &[MemoryBarrier], buffer_memory_barriers: &[BufferMemoryBarrier], image_memory_barriers: &[ImageMemoryBarrier]);
}

5. Error Handling

pub enum VkError {
    OutOfHostMemory, OutOfDeviceMemory, InitializationFailed, DeviceLost,
    MemoryMapFailed, LayerNotPresent, ExtensionNotPresent, FeatureNotPresent,

    Unknown(i32), Loader(Text),
}
