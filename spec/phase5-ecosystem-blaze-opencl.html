Blaze Phase 5 – Ecosystem: OpenCL

Previous: CUDA | Next: Appendix A

Goal

Low‑level, unsafe, data‑oriented bindings to OpenCL 3.0 heterogeneous computing API. Wraps platform, device, context, command queue, memory objects (buffer, image), program, kernel, event.

1. Core Types

pub struct Platform { id: cl_platform_id }
pub struct Device { id: cl_device_id }
pub struct Context { inner: cl_context }
pub struct CommandQueue { inner: cl_command_queue }
pub struct Event { inner: cl_event }
pub struct Buffer<T> { inner: cl_mem, size: usize, phantom: PhantomData<T> }
pub struct Image { inner: cl_mem, format: ImageFormat, width: usize, height: usize, depth: usize }
pub struct Sampler { inner: cl_sampler }
pub struct Program { inner: cl_program }
pub struct Kernel { inner: cl_kernel }

2. Platform and Device

impl Platform { pub fn enumerate() -> Result<Vec<Self>, OpenCLError>; pub fn devices(&self, device_type: DeviceType) -> Result<Vec<Device>, OpenCLError>; }
impl Device { pub fn name(&self) -> Result<Text, OpenCLError>; pub fn vendor(&self) -> Result<Text, OpenCLError>; pub fn version(&self) -> Result<Text, OpenCLError>; }

3. Context and Queue

impl Context { pub fn new(devices: &[&Device]) -> Result<Self, OpenCLError>; pub fn create_queue(&self, device: &Device, properties: QueueProperties) -> Result<CommandQueue, OpenCLError>; }
impl CommandQueue { pub fn finish(&self) -> Result<(), OpenCLError>; pub fn flush(&self) -> Result<(), OpenCLError>; }

4. Memory Objects

impl<T> Buffer<T> {
    pub fn size(&self) -> usize;
    pub unsafe fn enqueue_write(&self, queue: &CommandQueue, blocking: bool, offset: usize, data: &[T], events_wait: &[&Event]) -> Result<Event, OpenCLError>;
    pub unsafe fn enqueue_read(&self, queue: &CommandQueue, blocking: bool, offset: usize, data: &mut [T], events_wait: &[&Event]) -> Result<Event, OpenCLError>;
}

5. Program and Kernel

impl Program {
    pub fn create_program_with_source(context: &Context, source: &str) -> Result<Self, OpenCLError>;
    pub fn build(&self, devices: &[&Device], options: Option<&str>) -> Result<(), OpenCLError>;
    pub fn create_kernel(&self, name: &str) -> Result<Kernel, OpenCLError>;
}
impl Kernel {
    pub unsafe fn set_arg<T: Sized>(&self, index: u32, arg: &T) -> Result<(), OpenCLError>;
    pub unsafe fn set_arg_buffer<T>(&self, index: u32, buffer: &Buffer<T>) -> Result<(), OpenCLError>;
    pub unsafe fn enqueue_nd_range(&self, queue: &CommandQueue, global_work_size: &[usize], local_work_size: Option<&[usize]>, events_wait: &[&Event]) -> Result<Event, OpenCLError>;
}

6. Error Handling

pub enum OpenCLError {
    DeviceNotFound, DeviceNotAvailable, CompilerNotAvailable,
    OutOfHostMemory, OutOfResources, OutOfMemory, InvalidValue,
    InvalidDevice, InvalidBufferSize, InvalidGlobalWorkSize,
    InvalidWorkGroupSize, InvalidWorkItemSize, InvalidProgram,
    InvalidProgramExecutable, InvalidKernel, InvalidKernelArgs,
    InvalidCommandQueue, InvalidContext, InvalidEvent, InvalidOperation,
    BuildProgramFailure, MapFailure, UnmapFailure, ProfilingInfoNotAvailable,
    Unknown(i32), Loader(Text),
}
