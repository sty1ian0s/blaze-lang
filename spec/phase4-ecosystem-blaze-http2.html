Goal

HTTP/2 client and server with binary framing, multiplexing, HPACK compression, server push, flow control.

1. Frames

pub enum Frame {
    Data { stream_id: u32, flags: DataFlags, payload: Vec<u8> },
    Headers { stream_id: u32, flags: HeadersFlags, header_block_fragment: Vec<u8> },
    Priority { stream_id: u32, exclusive: bool, stream_dep: u32, weight: u8 },
    RstStream { stream_id: u32, error_code: ErrorCode },
    Settings { flags: SettingsFlags, settings: Vec<(SettingId, u32)> },
    PushPromise { stream_id: u32, promised_stream_id: u32, header_block_fragment: Vec<u8> },
    Ping { flags: PingFlags, opaque_data: [u8; 8] },
    GoAway { last_stream_id: u32, error_code: ErrorCode, debug_data: Vec<u8> },
    WindowUpdate { stream_id: u32, window_size_increment: u32 },
    Continuation { stream_id: u32, header_block_fragment: Vec<u8> },
}
