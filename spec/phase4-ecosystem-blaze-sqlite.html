Goal

SQLite driver implementing blaze‑sql traits. FFI to libsqlite3, async via blocking thread pool.

1. Database Type

pub struct Sqlite;
impl Database for Sqlite { type Connection = SqliteConnection; fn connect(uri: &str) -> Result<Self::Connection, Error>; }

2. Connection

pub struct SqliteConnection { db: *mut sqlite3, is_open: bool }
impl Connection for SqliteConnection {
    async fn execute(&self, query: &Query) -> Result<ExecuteResult, Error>;
    async fn query(&self, query: &Query) -> Result<Rows, Error>;
    async fn transaction(&self) -> Result<Transaction<'_, Self>, Error>;
    fn is_autocommit(&self) -> bool;
    fn set_autocommit(&mut self, value: bool) -> Result<(), Error>;
    fn close(self) -> Result<(), Error>;
}

3. Transaction

impl<'a> SqliteTransaction<'a> {
