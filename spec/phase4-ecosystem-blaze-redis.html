Blaze Phase 4 – Ecosystem: Redis

Previous: RabbitMQ | Next: Serde

Goal

Async Redis client with RESP2/RESP3, pipelining, transactions, Pub/Sub, cluster‑aware routing, connection pool.

1. RedisClient

pub struct RedisClient { stream: TcpStream, buf: Vec<u8>, state: ClientState }
impl RedisClient {
    pub async fn connect(uri: &str) -> Result<Self, RedisError>;
    pub async fn execute(&mut self, cmd: &Cmd) -> Result<RedisValue, RedisError>;
    pub async fn pipeline(&mut self, cmds: &[Cmd]) -> Result<Vec<RedisValue>, RedisError>;
    pub async fn subscribe(self, channels: &[Text]) -> Result<PubSub, RedisError>;
}

2. Cmd Builder

pub struct Cmd { args: Vec<RedisValue> }
impl Cmd {
    pub fn new(command: &str) -> Self;
    pub fn arg<T: ToRedisValue>(&mut self, value: T) -> &mut Self;
}
macro_rules! cmd { ($cmd:literal, $($arg:expr),*) => { … } }

3. RedisValue

pub enum RedisValue {
    Nil,
    SimpleString(Text),
    BulkString(Vec<u8>),
    Integer(i64),
    Array(Vec<RedisValue>),
    Double(f64),
    Boolean(bool),
    BigNumber(Vec<u8>),
    Error(RedisError),
}

4. RedisPool

pub struct RedisPool { /* actor handle */ }

5. PubSub

pub struct PubSub { receiver: Receiver<PubSubMessage> }
pub enum PubSubMessage {
    Message { channel: Text, pattern: Option<Text>, payload: Vec<u8> },
    Subscribe { channel: Text, count: usize },
    Unsubscribe { channel: Text, count: usize },
}

6. Error Handling

pub enum RedisError {
    Io(std::io::Error),
    Protocol(Text),
    Server(Text),
    AuthenticationFailed,
    ConnectionClosed,
    Timeout,
    PoolExhausted,
    TypeMismatch(Text),
}
