Blaze Phase 5 – Ecosystem: CUDA

Previous: Plot | Next: OpenCL

Goal

Low‑level, unsafe, data‑oriented bindings to NVIDIA CUDA GPU computing platform (Driver API v12.x). Wraps resources (device, context, stream, event, memory, module, function) in linear types.

1. Core Types

pub struct Device { id: i32 }
impl Device { pub fn count() -> usize; pub fn get(index: usize) -> Option<Self>; }
pub struct Context { inner: CUcontext }
impl Context { pub fn set_current(&self) -> Result<(), CudaError>; pub fn pop_current() -> Result<(), CudaError>; pub fn create_stream(&self) -> Result<Stream, CudaError>; }
pub struct Stream { inner: CUstream }
pub struct Event { inner: CUevent }
pub struct DeviceMemory { ptr: *mut u8, size: usize }
pub struct UnifiedMemory { ptr: *mut u8, size: usize }
pub struct Module { inner: CUmodule }
pub struct Function { inner: CUfunction }

2. Memory

impl DeviceMemory {
    pub fn new(size: usize) -> Result<Self, CudaError>;
    pub unsafe fn as_ptr(&self) -> *mut u8;
    pub fn size(&self) -> usize;
    pub unsafe fn copy_to_host(&self, host: *mut u8, size: usize) -> Result<(), CudaError>;
    pub unsafe fn copy_from_host(&self, host: *const u8, size: usize) -> Result<(), CudaError>;
    pub unsafe fn memset(&self, value: u8, size: usize) -> Result<(), CudaError>;
}
impl UnifiedMemory {
    pub fn new(size: usize) -> Result<Self, CudaError>;
}

3. Module and Kernel

impl Module {
    pub fn from_ptx(ptx: &str) -> Result<Self, CudaError>;
    pub fn from_file(path: &str) -> Result<Self, CudaError>;
    pub fn get_function(&self, name: &str) -> Result<Function, CudaError>;
}
impl Function {
    pub unsafe fn launch(&self, grid: (u32, u32, u32), block: (u32, u32, u32), shared_mem: u32, stream: &Stream, args: &[*mut c_void]) -> Result<(), CudaError>;
    pub fn max_threads_per_block(&self) -> i32;
    pub fn optimal_block_size(&self) -> i32;
}

4. Error Handling

pub enum CudaError {
    InvalidValue, OutOfMemory, NotInitialized, Deinitialized, NoDevice,
    InvalidDevice, InvalidImage, InvalidContext, ContextAlreadyCurrent,
    MapFailed, UnmapFailed, ArrayIsMapped, AlreadyMapped, NoBinaryForGpu,
    AlreadyAcquired, NotMapped, InvalidSource, FileNotFound, NvlinkFailed,
    InvalidPtx, UnloadingFailed, LaunchFailed, LaunchOutOfResources,
    LaunchTimeout, LaunchIncompatibleTexturing, Unknown(i32),
}
